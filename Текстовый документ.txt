DB_PATH = "bot.db"

def init_db():
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute('''CREATE TABLE IF NOT EXISTS users (
        tg_id INTEGER PRIMARY KEY,
        username TEXT,
        sub_until INTEGER DEFAULT 0,
        balance REAL DEFAULT 100,
        total_bet REAL DEFAULT 0,
        total_win REAL DEFAULT 0
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS tg_accounts (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        owner_tg_id INTEGER,
        phone TEXT,
        session_file TEXT,
        is_active INTEGER DEFAULT 1
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS vk_accounts (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        owner_tg_id INTEGER,
        token TEXT,
        vk_user_id INTEGER,
        vk_name TEXT
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS withdraw_requests (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER,
        amount REAL,
        wallet TEXT,
        status TEXT DEFAULT 'pending',
        created_at INTEGER
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS templates (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER,
        name TEXT,
        content TEXT
    )''')
    conn.commit()
    conn.close()

def get_user(tg_id):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT tg_id, username, sub_until, balance, total_bet, total_win FROM users WHERE tg_id=?", (tg_id,))
    row = c.fetchone()
    conn.close()
    if row:
        return {"tg_id": row[0], "username": row[1], "sub_until": row[2] or 0, "balance": row[3], "total_bet": row[4], "total_win": row[5]}
    return None

def create_user(tg_id, username):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("INSERT OR IGNORE INTO users (tg_id, username, sub_until, balance, total_bet, total_win) VALUES (?, ?, 0, 100, 0, 0)", (tg_id, username))
    conn.commit()
    conn.close()

def is_subscribed(tg_id):
    user = get_user(tg_id)
    if not user: return False
    return user["sub_until"] > int(time.time())

def set_subscription(tg_id, days):
    new_time = int(time.time()) + days * 86400
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("UPDATE users SET sub_until=? WHERE tg_id=?", (new_time, tg_id))
    conn.commit()
    conn.close()

CRYPTOBOT_API_URL = "https://pay.crypt.bot/api"

async def create_crypto_invoice(amount_usd: float, description: str):
    url = f"{CRYPTOBOT_API_URL}/createInvoice"
    headers = {"Crypto-Pay-API-Token": CRYPTOBOT_TOKEN}
    payload = {
        "asset": "USDT",
        "amount": str(amount_usd),
        "description": description,
        "paid_btn_name": "callback",
        "paid_btn_url": "https://t.me/YOUR_BOT_USERNAME"  # замените
    }
    try:
        async with aiohttp.ClientSession() as session:
            async with session.post(url, headers=headers, json=payload) as resp:
                data = await resp.json()
                if data.get("ok"):
                    return {"pay_url": data["result"]["pay_url"], "invoice_id": data["result"]["invoice_id"]}
                return None
    except:
        return None

async def check_crypto_invoice(invoice_id: str):
    url = f"{CRYPTOBOT_API_URL}/getInvoices"
    headers = {"Crypto-Pay-API-Token": CRYPTOBOT_TOKEN}
    params = {"invoice_ids": invoice_id}
    try:
        async with aiohttp.ClientSession() as session:
            async with session.get(url, headers=headers, params=params) as resp:
                data = await resp.json()
                if data.get("ok") and data["result"]["items"]:
                    return data["result"]["items"][0]["status"]
                return None
    except:
        return None

@dp.callback_query(F.data.startswith("broadcast_target_"))
async def start_broadcast(callback: CallbackQuery, state: FSMContext):
    target = callback.data.split("_")[2]
    data = await state.get_data()
    text = data["broadcast_text"]
    delay = data["delay"]
    user_id = callback.from_user.id
    if target == "tg":
        tg_acc = get_active_tg_account(user_id)
        if not tg_acc:
            await callback.answer("❌ Сначала подключите Telegram аккаунт", show_alert=True)
            return
        await callback.message.answer("🚀 Начинаю рассылку в Telegram...")
        await run_tg_broadcast(user_id, text, delay, callback.message.chat.id)
    elif target == "vk":
        vk_acc = get_active_vk_account(user_id)
        if not vk_acc:
            await callback.answer("❌ Сначала подключите VK аккаунт", show_alert=True)
            return
        await callback.message.answer("🚀 Начинаю рассылку в VK...")
        await run_vk_broadcast(user_id, text, delay, callback.message.chat.id)
    await state.clear()
    await callback.answer()

async def run_tg_broadcast(user_id, text, delay, chat_id):
    tg_acc = get_active_tg_account(user_id)
    if not tg_acc:
        await bot.send_message(chat_id, "❌ Аккаунт не найден")
        return
    client = TelegramClient(tg_acc["session_file"], API_ID, API_HASH)
    await client.connect()
    dialogs = await client.get_dialogs()
    targets = [d for d in dialogs if d.is_user or d.is_group or d.is_channel]
    total = len(targets)
    await bot.send_message(chat_id, f"📨 Найдено {total} получателей. Задержка {delay} сек.")
    sent = 0
    for i, dialog in enumerate(targets):
        try:
            await client.send_message(dialog.entity, text)
            sent += 1
            await asyncio.sleep(delay)
            if (i+1) % 10 == 0:
                await bot.send_message(chat_id, f"📊 Прогресс: {sent}/{total}")
        except:
            pass
    await client.disconnect()
    await bot.send_message(chat_id, f"✅ Рассылка завершена. Отправлено {sent} из {total}.")

async def run_vk_broadcast(user_id, text, delay, chat_id):
    vk_acc = get_active_vk_account(user_id)
    if not vk_acc:
        await bot.send_message(chat_id, "❌ Аккаунт не найден")
        return
    vk_session = vk_api.VkApi(token=vk_acc["token"])
    vk = vk_session.get_api()
    friends = vk.friends.get()["items"]
    convs = vk.messages.getConversations(count=200)["items"]
    targets = friends + [c["conversation"]["peer"]["id"] for c in convs]
    total = len(targets)
    await bot.send_message(chat_id, f"📨 Найдено {total} получателей. Задержка {delay} сек.")
    sent = 0
    for i, target in enumerate(targets):
        try:
            if isinstance(target, int):
                vk.messages.send(user_id=target, message=text, random_id=0)
            else:
                vk.messages.send(peer_id=target, message=text, random_id=0)
            sent += 1
            await asyncio.sleep(delay)
            if (i+1) % 10 == 0:
                await bot.send_message(chat_id, f"📊 Прогресс: {sent}/{total}")
        except:
            pass
    await bot.send_message(chat_id, f"✅ Рассылка завершена. Отправлено {sent} из {total}.")

# --- Вспомогательные функции для кубиков с анимацией ---
async def throw_dice(chat_id, count=1):
    values = []
    for _ in range(count):
        msg = await bot.send_dice(chat_id, emoji="🎲")
        values.append(msg.dice.value)
        if count > 1:
            await asyncio.sleep(0.6)
    return values

async def play_1cube(user_id, bet, mult, chat_id):
    roll = (await throw_dice(chat_id, 1))[0]
    coeff = 1.5 if mult == "turbo" else int(mult)
    win = roll >= 4
    if win:
        payout = bet * coeff
        update_balance(user_id, payout, bet, payout)
        return f"🎲 Выпало {roll}\n✅ Выигрыш: {bet:.2f} x{coeff} = {payout:.2f} 🎲"
    else:
        update_balance(user_id, -bet, bet, 0)
        return f"🎲 Выпало {roll}\n❌ Проигрыш: {bet:.2f} 🎲"

async def show_mines_field(message: types.Message, user_id):
    game = mines_games.get(user_id)
    if not game:
        return
    keyboard = []
    for i in range(9):
        if i in game["opened"]:
            if i in game["mines"]:

            else:
                symbol = "✅"
        else:
            symbol = "⬜"
        row = i // 3
        if len(keyboard) <= row:

    keyboard.append([InlineKeyboardButton(text="🏆 Забрать выигрыш", callback_data="mines_cashout")])

@dp.callback_query(F.data.startswith("mines_cell_"))
async def mines_open_cell(callback: CallbackQuery):
    user_id = callback.from_user.id
    if user_id not in mines_games:
        await callback.answer("Игра не найдена", show_alert=True)
        return
    game = mines_games[user_id]
    cell = int(callback.data.split("_")[2])
    if cell in game["opened"]:
        await callback.answer("Клетка уже открыта", show_alert=True)
        return
    if cell in game["mines"]:
        # Проигрыш
        update_balance(user_id, -game["bet"], game["bet"], 0)
        await callback.message.edit_text(f"💣 *БАХ!* Вы наступили на мину.\n❌ Проигрыш: {game['bet']:.2f} 🎲", parse_mode="Markdown", reply_markup=back_button("casino_main"))
        del mines_games[user_id]
        await callback.answer()
        return
    # Открываем безопасную клетку
    game["opened"].add(cell)
    safe_opened = len([c for c in game["opened"] if c not in game["mines"]])
    total_safe = len(game["safe_cells"])
    if safe_opened == total_safe:
        # Выигрыш
        payout = game["bet"] * game["multiplier"]
        update_balance(user_id, payout, game["bet"], payout)
        await callback.message.edit_text(f"🎉 *ПОБЕДА!* Вы открыли все безопасные клетки!\n✅ Выигрыш: {game['bet']:.2f} x{game['multiplier']} = {payout:.2f} 🎲", parse_mode="Markdown", reply_markup=back_button("casino_main"))
        del mines_games[user_id]
        await callback.answer()
        return
    # Обновляем поле
    await show_mines_field(callback.message, user_id)
    await callback.answer()

@dp.callback_query(F.data == "mines_cashout")
async def mines_cashout(callback: CallbackQuery):
    user_id = callback.from_user.id
    if user_id not in mines_games:
        await callback.answer("Игра не найдена", show_alert=True)
        return
    game = mines_games[user_id]
    safe_opened = len([c for c in game["opened"] if c not in game["mines"]])
    if safe_opened == 0:
        await callback.answer("Нет открытых клеток, вы не можете забрать выигрыш", show_alert=True)
        return
    # Выигрыш пропорционально открытым клеткам (упрощённо)
    total_safe = len(game["safe_cells"])
    coeff = (safe_opened / total_safe) * game["multiplier"]
    payout = game["bet"] * coeff
    update_balance(user_id, payout, game["bet"], payout)
    await callback.message.edit_text(f"💰 Вы забрали выигрыш досрочно.\n✅ Выигрыш: {payout:.2f} 🎲", parse_mode="Markdown", reply_markup=back_button("casino_main"))
    del mines_games[user_id]
    await callback.answer()

        except: pass
    await status_msg.edit_text(f"✅ Отправлено {sent} пользователям.")
    await state.clear()

@dp.callback_query(F.data == "admin_stats")

@dp.callback_query(F.data == "admin_withdraws")
async def admin_withdraws(callback: CallbackQuery):
    if callback.from_user.id != ADMIN_ID: return
    pending = get_pending_withdraws()
    if not pending:
        await callback.message.edit_text("📭 Нет заявок", reply_markup=back_button("admin_panel"))
        await callback.answer()
        return
    for req in pending:
        req_id, user_id, amount, wallet, _ = req
        text = f"💰 Заявка #{req_id}\n👤 {user_id}\n💵 {amount:.2f}\n💳 {wallet}"
        kb = InlineKeyboardMarkup(inline_keyboard=[
            [InlineKeyboardButton(text="✅ Одобрить", callback_data=f"approve_{req_id}"),
             InlineKeyboardButton(text="❌ Отклонить", callback_data=f"reject_{req_id}")]
        ])
        await callback.message.answer(text, reply_markup=kb)
    await callback.message.delete()
    await callback.answer()

@dp.callback_query(F.data.startswith("approve_"))
async def approve_withdraw(callback: CallbackQuery):
    req_id = int(callback.data.split("_")[1])
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT user_id, amount FROM withdraw_requests WHERE id=?", (req_id,))
    row = c.fetchone()
    if row:
        user_id, amount = row
        current = get_balance(user_id)
        if current >= amount:
            set_balance(user_id, current - amount)
            update_withdraw_status(req_id, "approved")
            await bot.send_message(user_id, f"✅ Вывод {amount:.2f} одобрен")
            await callback.message.edit_text(f"✅ Заявка #{req_id} одобрена")
        else:
            await callback.message.edit_text(f"❌ У пользователя недостаточно средств")
    else:
        await callback.message.edit_text("❌ Заявка не найдена")
    conn.close()
    await callback.answer()

@dp.callback_query(F.data.startswith("reject_"))
async def reject_withdraw(callback: CallbackQuery):
    req_id = int(callback.data.split("_")[1])
    update_withdraw_status(req_id, "rejected")
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT user_id FROM withdraw_requests WHERE id=?", (req_id,))
    row = c.fetchone()
    if row:
        await bot.send_message(row[0], "❌ Заявка на вывод отклонена")
    conn.close()
    await callback.message.edit_text(f"❌ Заявка #{req_id} отклонена")
    await callback.answer()
